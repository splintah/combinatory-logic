module CombinatoryLogic.Forest where

forest : Forest
forest = record
  { Bird = Combinator
  ; _≈_ = _≈_
  ; _∙_ = Syntax._∙_
  ; isForest = record
    { isEquivalence = isEquivalence
    ; ∙-cong = ∙-cong
    }
  }

open Forest forest
